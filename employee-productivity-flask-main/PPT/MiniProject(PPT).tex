% -----------------------------------------------------
\begin{frame}{\textbf{Introduction}}
\justify
\textbf{Objective:}  
To develop a machine learning model for predicting employee attrition and performance using HR-related data.

\medskip
\textbf{Focus:}  
Analyzing significant patterns, feature importance, and predictive insights from employee behavioral and performance datasets.

\medskip
\textbf{Outcome:}  
Enables data-driven retention strategies, improved workforce planning, reduced attrition, and enhanced organizational productivity.
\end{frame}

% -----------------------------------------------------
\begin{frame}{\textbf{Methodology}}
\justify
\begin{enumerate}
    \item \textbf{Data Collection:} Demographics, job role, experience data.
    \item \textbf{Preprocessing:} Handling missing values, removing outliers, scaling.
    \item \textbf{Feature Selection:} Identifying key attributes using feature importance.
    \item \textbf{Model Development:} Random Forest, XGBoost.
    \item \textbf{Evaluation:} Accuracy, Precision, Recall, F1-Score.
    \item \textbf{Deployment:} Predictive dashboard for HR insights.
\end{enumerate}
\end{frame}

\begin{frame}{\textbf{Limitations}}
\justify
\textbf{Data \& Features}
\begin{itemize}
    \item Limited/imbalanced datasets reduce accuracy.
    \item Missing behavioral or psychological factors restrict depth.
\end{itemize}

\textbf{System \& Usability}
\begin{itemize}
    \item Black-box ML models reduce interpretability.
    \item Privacy concerns due to sensitive employee data.
    \item Requires continuous retraining as policies evolve.
\end{itemize}
\end{frame}
